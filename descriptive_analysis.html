Retention of customers monthly

# Extract Year and Month in numeric format for proper ordering
online_sales$Month <- format(online_sales$Transaction_Date, "%b")

# Get unique customer transactions by month
data_retention <- online_sales %>%
  arrange(CustomerID, Month) %>%
  group_by(CustomerID, Month) %>%
  summarise(Transactions = n(), .groups = "drop") %>%
  mutate(
    Prev_Month = lag(Month), 
    Retained = if_else(!is.na(Prev_Month) & Month > Prev_Month, 1, 0) 
  )

# Summarize retention rates by month
monthly_retention <- data_retention %>%
  group_by(Month) %>%
  summarise(
    Retention_Rate = sum(Retained) / n_distinct(CustomerID),
    .groups = "drop"
  )

# Plot retention rates
ggplot(monthly_retention, aes(x = Month, y = Retention_Rate, group = 1)) +
  geom_line(color = "blue", linewidth = 1) +
  theme_minimal() +
  labs(
    title = "Monthly Retention ",
    x = "Month",
    y = "Rate of Retention(%)"
  ) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

Monthly Revenue from existing vs.new customers

# Categorize customers (New/Existing)
online_sales <- online_sales %>%
  arrange(CustomerID, Transaction_Date) %>%
  group_by(CustomerID) %>%
  mutate(Customer_Type = if_else(row_number() == 1, "New", "Existing"))

# Monthly revenue by customer type
revenue_by_type <- online_sales %>%
  group_by(Month, Customer_Type) %>%
  summarise(Revenue = sum(Online_Spend)) %>%
  ungroup()
## `summarise()` has grouped output by 'Month'. You can override using the
## `.groups` argument.
# Plot the results
ggplot(revenue_by_type, aes(x = Month, y = Revenue, fill = Customer_Type)) +
  geom_bar(stat = "identity", position = "dodge") +
  theme_minimal() +
  labs(title = "Monthly Revenue per Customer type", x = "Month", y = "Revenue") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
